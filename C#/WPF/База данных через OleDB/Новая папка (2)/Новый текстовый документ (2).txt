CREATE TABLE Zakaz
(
	[idZakaz] INT NOT NULL PRIMARY KEY, 
    [idPokup] INT NULL FOREIGN KEY REFERENCES Pokupatel(idPokup), 
    [idTovar] INT NULL FOREIGN KEY REFERENCES Tovar(idTovar), 
    [amount] INT NULL, 
    [date] DATETIME NULL
)
CREATE TABLE Tovar
(
	[idTovar] INT NOT NULL PRIMARY KEY, 
    [name] NCHAR(10) NULL, 
    [amount] INT NULL, 
    [price] INT NULL
)

CREATE TABLE Pokupatel
(
	[idPokup] INT NOT NULL PRIMARY KEY, 
    [name] NCHAR(10) NULL
)
