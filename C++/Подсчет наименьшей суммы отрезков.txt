int main()
{
const int N=100;
int a[N];
int k,t,s,n;

cout<<"Chislo n=";
	cin>>n;
cout<<"Massiv: ";
	cout<<endl;

	for (int i=0;i<n;i++)//Ввод массива.
	{
		cin>>a[i];

		for (int i=0;i<n;i++)//Вывод массива.
	{
		cout<<a[i]<<" ";
	}
cout<<endl;

s=0;
for (int i=0;i<n;i++)//Цикл подсчета отрезков для самого первого элемента
	{
		if (a[0]>=0 && a[i]>=0) s=s+abs(a[0]-a[i]);
			if (a[0]>=0 && a[i]<0) s=s+abs(a[0])+abs(a[i]);
		if (a[0]<=0 && a[i]<0) s=s+abs(a[0]-a[i]);
			if (a[0]<=0 && a[i]>0)	s=s+abs(a[0])+abs(a[i]);
	}
t=0;

for (int i=1; i<n; i++)//Цикл выбора наименьшей суммы
	{
		for (int j=0; j<n; j++)
		{
		if (a[i]>=0 && a[j]>=0) k=k+abs(a[0]-a[i]);
			if (a[i]>=0 && a[j]<0) k=k+abs(a[0])+abs(a[i]);
		if (a[i]<=0 && a[j]<0) k=k+abs(a[0]-a[i]);
			if (a[i]<=0 && a[j]>0)	k=k+abs(a[0])+abs(a[i]);
		}

		{		
		if (s>k)
			{
				s=k;
				t=a[i];
			}
		}//В противном случае S-остаётся прежним и берется новый элемент I-тый из массива "а".
		k=0;
	}


cout<<endl;
cout<<"Minimalnaya summa: tochka "<<t<<" s summoi = "<<s<<endl;

}
