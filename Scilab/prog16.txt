//Программа находит номера минимальных элементов в массиве
clc
function [r] = nummin(A, B)
    n = length(A)//Длина первого массива
    mmin = A(1)//Значение минимального элемента в первом массиве
    for i = 2:n
        if A(i) < mmin then
            mmin = A(i)
        end
    end
    j = 1
    for i = 1:n
        if A(i) == mmin then
            r(1,j) = i//Запоминание индекса
            j = j + 1
        end
    end
    n = length(B)//Длина второго массива
    mmin = B(1)//Значение минимального элемента во втором массиве
    for i = 2:n
        if B(i) < mmin then
            mmin = B(i)
        end
    end
    j = 1
    for i = 1:n
        if B(i) == mmin then
            r(2,j) = i//Запоминание индекса
            j = j + 1
        end
    end
endfunction
A = [86 5 4 3 6 57 8];//Объявление первого массива
B = [874 34 498 5 4 845 34 5];//Объявление второго массива
nummin(A,B)
